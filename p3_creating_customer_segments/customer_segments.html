abnormal_data_dic = {}
for feature in log_data.keys():
    feature_log = log_data[feature]
    
    # TODOQ125th
    Q1 = np.percentile(feature_log, 25)
    
    # TODOQ375th
    Q3 = np.percentile(feature_log, 75)
    
    # TODO1.5
    step = 1.5 * (Q3 - Q1)
    
    # 
#     print "Data points considered outliers for the feature '{}':".format(feature)
#     display(log_data[~((log_data[feature] >= Q1 - step) & (log_data[feature] <= Q3 + step))])
    abnormal_data = (log_data[~((log_data[feature] >= Q1 - step) & (log_data[feature] <= Q3 + step))]).index
    for i in abnormal_data:
        if i not in abnormal_data_dic:
            abnormal_data_dic[i] = 1
        else:
            abnormal_data_dic[i] += 1
